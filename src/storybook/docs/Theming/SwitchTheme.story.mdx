const themeMap = {
    light: {}, // that theme you want to overwrite default theme
    dark: RcDefaultDarkTheme // that theme you want to overwrite default theme
}

<RcThemeSwitcherProvider defaultTheme="light" themeMap={themeMap}>
  <RcThemeProvider>
    <ThemeSwitcher />

  </RcThemeProvider>

### useThemeSwitcher

use `useThemeSwitcher` to set theme or get theme value from hook.

- ThemeSwitcher

```tsx
import { RcButton, useThemeSwitcher } from '@ringcentral/juno';
import React, { FunctionComponent } from 'react';

export interface ThemeSwitcherProps {}

export const ThemeSwitcher: FunctionComponent<ThemeSwitcherProps> = () => {
  const { setTheme, themeMap } = useThemeSwitcher();

  return (
    <>
      {Object.keys(themeMap).map((key) => (
        <RcButton
          onClick={() => {
            setTheme(key);
          }}
        >
          {key}
        </RcButton>
      ))}
    </>
  );
};
```
